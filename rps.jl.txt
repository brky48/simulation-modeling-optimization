using JuMP, HiGHS

m = Model(HiGHS.Optimizer)

@variable(m, g)
@variable(m, p[1:3] >= 0)

@objective(m, Max, g)

@constraint(m, sum(p) == 1)
# @constraint(m, p[1] +  p[2] + p[3] == 1)
@constraint(m, 0*p[1] + (-1)*p[2] + 1*p[3] >= g)
@constraint(m, 1*p[1] + 0*p[2] + (-1)*p[3] >= g)
@constraint(m, (-1)*p[1] + 1*p[2] + 0*p[3] >= g)

@constraint(m, p[1] <= 1)
@constraint(m, p[2] <= 1)
@constraint(m, p[3] <= 1)

optimize!(m)
